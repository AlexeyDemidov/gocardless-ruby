def source=(obj)
  klass = obj.class.to_s.split(':').last
  if !%w{Subscription PreAuthorization}.include?(klass)
    raise ArgumentError, ("Object must be an instance of Subscription or "
                          "PreAuthorization")
  end
  @source_id = obj.id
  @source_type = klass.underscore
end
